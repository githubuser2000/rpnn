\section*{Beispiel 1: Rechnungen in den $3$-adischen Zahlen}

Wir betrachten die Elemente
\[
x = 2 + 3 + 3^2
\qquad\text{und}\qquad
y = 1 + 2\cdot 3 + 3^2 .
\]

\subsection*{Addition}
\[
x+y
=
(2+1)
+
(1+2)\cdot 3
+
(1+1)\cdot 3^2
=
3 + 3\cdot 3 + 2\cdot 3^2 .
\]
Da $3 = 1\cdot 3 + 0$ entsteht ein Übertrag:
\[
x+y
=
0 + 1\cdot 3 + 3\cdot 3 + 2\cdot 3^2
=
0 + 1\cdot 3 + 0\cdot 3^2 + 3\cdot 3^2 .
\]

\subsection*{Subtraktion}
\[
x-y
=
(2-1)
+
(1-2)\cdot 3
+
(1-1)\cdot 3^2
=
1 - 3 .
\]
Borrow-Term:
\[
1-3 = 1+3^1\!\cdot (-1)
=
1 + 2\cdot 3 .
\]

\subsection*{Multiplikation}
\[
x=\sum_{i\ge 0}a_i3^i,\quad y=\sum_{j\ge 0}b_j3^j,
\]
\[
xy=\sum_{n\ge0}\Bigl(\sum_{i+j=n}a_ib_j\Bigr)3^n.
\]
Koeffizienten:
\[
a_0=2,\ a_1=1,\ a_2=1,
\qquad
b_0=1,\ b_1=2,\ b_2=1.
\]
\[
\Rightarrow\quad
\begin{aligned}
n=0:&\quad 2\cdot 1 = 2\\
n=1:&\quad 2\cdot2 + 1\cdot1 = 5 = 1 + 1\cdot 3\\
n=2:&\quad 2\cdot1 + 1\cdot2 + 1\cdot1 = 5 = 1 + 1\cdot 3\\
\end{aligned}
\]
Damit
\[
xy
=
2
+
1\cdot 3
+
1\cdot 3^2
+
\text{Überträge}.
\]

\subsection*{Division}
Sei
\[
y = 1+2\cdot 3 = 1+z,\qquad v_3(z)>0.
\]
Dann gilt die Neumannreihe
\[
\frac{1}{1+z}
=
\sum_{n=0}^{\infty} (-z)^n
=
1 - z + z^2 - z^3 + \cdots
\]
also
\[
\frac{1}{y}
=
1 - 2\cdot 3 + 4\cdot 3^2 - 8\cdot 3^3 + \cdots .
\]

\section*{Beispiel 2: Rechnungen in den $5$-adischen Zahlen}

\[
u = 3 + 4\cdot 5 + 2\cdot 5^2,
\qquad
v = 1 + 3\cdot 5.
\]

\subsection*{Addition}
\[
u+v
=
(3+1)
+
(4+3)\cdot 5
+
2\cdot 5^2
=
4 + 7\cdot 5 + 2\cdot 5^2.
\]
Übertrag:
\[
7 = 1\cdot 5 + 2
\Rightarrow
u+v
=
4 + 2\cdot 5 + 1\cdot 5^2 + 2\cdot 5^2.
\]

\subsection*{Multiplikation}
\[
uv
=
3\cdot 1
+
(3\cdot 3 + 4\cdot 1)5
+
\cdots
=
3 + 13\cdot 5 + \cdots
=
3 + 3\cdot 5 + 2\cdot 5^2 + \cdots .
\]

\subsection*{Division über Einheitenzerlegung}
\[
v = 1+3\cdot 5 = 1+w,\qquad v_5(w)>0,
\]
\[
\frac{1}{v}
=
1 - w + w^2 - w^3 + \cdots
=
1 - 3\cdot 5 + 9\cdot 5^2 - 27\cdot 5^3 + \cdots .
\]

\[
\Rightarrow\quad
\frac{u}{v}
=
u\,
\bigl(1-3\cdot 5+9\cdot 5^2-\cdots\bigr).
\]
